```typescript
import * as ImagePicker from 'expo-image-picker';

export async function pickMedia(options?: {
  mediaTypes?: ImagePicker.MediaType[];
  videoMaxDuration?: number;
}): Promise<ImagePicker.ImagePickerAsset | null> {
  // Request permissions FIRST to avoid iOS double-dialog (see RESEARCH.md Pitfall 1)
  const { status } = await ImagePicker.requestMediaLibraryPermissionsAsync();
  if (status !== 'granted') return null;

  const result = await ImagePicker.launchImageLibraryAsync({
    mediaTypes: options?.mediaTypes ?? ['images', 'videos'],
    allowsEditing: false,
    quality: 0.7, // Compression fallback for Expo Go (STORY-07, per user decision)
    videoMaxDuration: options?.videoMaxDuration ?? 30,
  });

  if (result.canceled || !result.assets?.length) return null;
  return result.assets[0]; // Returns full asset with uri, type, duration, width, height
}
```

**2. `uploadVideoResumable` function:**
```typescript
import { Upload } from 'tus-js-client';

export function uploadVideoResumable(
  fileUri: string,
  bucket: string,
  path: string,
  accessToken: string,
  onProgress?: (percentage: number) => void,
): Promise<string> {
  return new Promise(async (resolve, reject) => {
    try {
      const response = await fetch(fileUri);
      const blob = await response.blob();

      // Extract project ID from EXPO_PUBLIC_SUPABASE_URL
      const supabaseUrl = process.env.EXPO_PUBLIC_SUPABASE_URL ?? '';
      const projectId = supabaseUrl.replace('https://', '').split('.')[0];

      const upload = new Upload(blob, {
        endpoint: `https://${projectId}.storage.supabase.co/storage/v1/upload/resumable`,
        retryDelays: [0, 3000, 5000, 10000, 20000],
        headers: {
          authorization: `Bearer ${accessToken}`,
          'x-upsert': 'true',
        },
        uploadDataDuringCreation: true,
        removeFingerprintOnSuccess: true,
        chunkSize: 6 * 1024 * 1024, // 6MB required by Supabase
        metadata: {
          bucketName: bucket,
          objectName: path,
          contentType: fileUri.endsWith('.mov') ? 'video/quicktime' : 'video/mp4',
          cacheControl: '3600',
        },
        onError: (error) => reject(error),
        onProgress: (bytesUploaded, bytesTotal) => {
          onProgress?.(Math.round((bytesUploaded / bytesTotal) * 100));
        },
        onSuccess: () => {
          const { data } = supabase.storage.from(bucket).getPublicUrl(path);
          resolve(data.publicUrl);
        },
      });

      // Resume previous upload if exists
      const previousUploads = await upload.findPreviousUploads();
      if (previousUploads.length) {
        upload.resumeFromPreviousUpload(previousUploads[0]);
      }

      upload.start();
    } catch (err) {
      reject(err);
    }
  });
}
```

Rewrite `useCreateStory` to accept a media asset (not just pick internally):

```typescript
type MediaAsset = {
  uri: string;
  type?: 'image' | 'video'; // from ImagePickerAsset
  duration?: number; // in seconds, from ImagePickerAsset
};

export function useCreateStory() {
  const { user } = useAuth();
  const [loading, setLoading] = useState(false);
  const [uploadProgress, setUploadProgress] = useState(0);

  const createStory = useCallback(async (asset: MediaAsset) => {
    if (!user) return null;
    setLoading(true);
    setUploadProgress(0);

    try {
      const isVideo = asset.type === 'video';
      const ext = isVideo ? (asset.uri.endsWith('.mov') ? 'mov' : 'mp4') : 'jpg';
      const storagePath = `${user.id}/${Date.now()}.${ext}`;
      let publicUrl: string | null = null;

      if (isVideo) {
        // Get fresh session token (avoid Pitfall 2: token expiry during long uploads)
        const { data: { session } } = await supabase.auth.getSession();
        if (!session?.access_token) throw new Error('No session');

        publicUrl = await uploadVideoResumable(
          asset.uri,
          'stories',
          storagePath,
          session.access_token,
          (pct) => setUploadProgress(pct),
        );
      } else {
        // Standard upload for images (small files)
        publicUrl = await uploadImage(asset.uri, 'stories', storagePath);
      }

      if (!publicUrl) { setLoading(false); return null; }

      const expiresAt = new Date(Date.now() + 24 * 60 * 60 * 1000).toISOString();
      const durationMs = isVideo && asset.duration ? Math.round(asset.duration * 1000) : null;

      const { data, error } = await supabase
        .from('stories')
        .insert({
          author_id: user.id,
          media_url: publicUrl,
          type: isVideo ? 'video' : 'image',
          expires_at: expiresAt,
          storage_path: storagePath,
          duration_ms: durationMs,
        })
        .select('id')
        .single();

      if (error) { console.error('Create story error:', error.message); setLoading(false); return null; }

      setLoading(false);
      setUploadProgress(0);
      return data;
    } catch (err) {
      console.error('Create story failed:', err);
      setLoading(false);
      setUploadProgress(0);
      return null;
    }
  }, [user]);

  return { createStory, loading, uploadProgress };
}
```
