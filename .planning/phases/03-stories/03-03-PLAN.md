**A. Create `hooks/useStoryViewer.ts`:**

State machine for the story viewer. Manages creator navigation, story index, progress animation, and pause state.

```typescript
import { useSharedValue, withTiming, cancelAnimation, Easing, runOnJS } from 'react-native-reanimated';
import { useCallback, useRef, useState } from 'react';

type StoryItem = {
  id: string;
  mediaUrl: string;
  type: 'image' | 'video';
  durationMs: number | null;
  isSeen: boolean;
};

type StoryGroup = {
  authorId: string;
  authorName: string;
  avatarUrl: string | null;
  stories: StoryItem[];
};

const PHOTO_DURATION = 5000; // 5 seconds for photos

export function useStoryViewer(groups: StoryGroup[], onClose: () => void) {
  // State
  const [creatorIndex, setCreatorIndex] = useState(0);
  const [storyIndex, setStoryIndex] = useState(0);
  const [isPaused, setIsPaused] = useState(false);
  const [mediaReady, setMediaReady] = useState(false);

  // Reanimated shared values
  const progress = useSharedValue(0);
  const remainingDuration = useRef(0);

  const currentGroup = groups[creatorIndex];
  const currentStory = currentGroup?.stories[storyIndex];

  // Get duration for current story
  const getDuration = useCallback(() => {
    if (!currentStory) return PHOTO_DURATION;
    if (currentStory.type === 'video' && currentStory.durationMs) {
      return currentStory.durationMs;
    }
    return PHOTO_DURATION;
  }, [currentStory]);

  // Start progress animation (call ONLY after media is ready)
  const startProgress = useCallback(() => {
    const duration = getDuration();
    remainingDuration.current = duration;
    progress.value = 0;
    progress.value = withTiming(1, {
      duration,
      easing: Easing.linear,
    }, (finished) => {
      if (finished) runOnJS(goToNextStory)();
    });
  }, [getDuration]);

  // Pause
  const pause = useCallback(() => {
    setIsPaused(true);
    remainingDuration.current = getDuration() * (1 - progress.value);
    cancelAnimation(progress);
  }, [getDuration, progress]);

  // Resume
  const resume = useCallback(() => {
    setIsPaused(false);
    progress.value = withTiming(1, {
      duration: remainingDuration.current,
      easing: Easing.linear,
    }, (finished) => {
      if (finished) runOnJS(goToNextStory)();
    });
  }, [progress]);

  // Navigate next story (within creator or to next creator)
  const goToNextStory = useCallback(() => {
    cancelAnimation(progress);
    setMediaReady(false);

    if (!currentGroup) { onClose(); return; }

    if (storyIndex < currentGroup.stories.length - 1) {
      // Next story in same creator
      progress.value = 0;
      setStoryIndex(i => i + 1);
    } else if (creatorIndex < groups.length - 1) {
      // Next creator
      progress.value = 0;
      setStoryIndex(0);
      setCreatorIndex(i => i + 1);
    } else {
      // End -- close viewer
      onClose();
    }
  }, [storyIndex, creatorIndex, currentGroup, groups.length, onClose, progress]);

  // Navigate previous story
  const goToPrevStory = useCallback(() => {
    cancelAnimation(progress);
    setMediaReady(false);

    if (storyIndex > 0) {
      progress.value = 0;
      setStoryIndex(i => i - 1);
    } else if (creatorIndex > 0) {
      // Go to previous creator's last story
      progress.value = 0;
      const prevGroup = groups[creatorIndex - 1];
      setStoryIndex(prevGroup.stories.length - 1);
      setCreatorIndex(i => i - 1);
    } else {
      // Restart current story
      progress.value = 0;
      setMediaReady(false);
    }
  }, [storyIndex, creatorIndex, groups, progress]);

  // Swipe to next/prev creator
  const goToNextCreator = useCallback(() => {
    cancelAnimation(progress);
    setMediaReady(false);
    if (creatorIndex < groups.length - 1) {
      progress.value = 0;
      setStoryIndex(0);
      setCreatorIndex(i => i + 1);
    } else {
      onClose();
    }
  }, [creatorIndex, groups.length, onClose, progress]);

  const goToPrevCreator = useCallback(() => {
    cancelAnimation(progress);
    setMediaReady(false);
    if (creatorIndex > 0) {
      progress.value = 0;
      setStoryIndex(0);
      setCreatorIndex(i => i - 1);
    }
  }, [creatorIndex, progress]);

  // Called when media finishes loading
  const onMediaReady = useCallback(() => {
    setMediaReady(true);
    startProgress();
  }, [startProgress]);

  // Called when video ends naturally
  const onVideoEnd = useCallback(() => {
    goToNextStory();
  }, [goToNextStory]);

  // Open viewer at specific creator
  const openAt = useCallback((groupIndex: number) => {
    setCreatorIndex(groupIndex);
    setStoryIndex(0);
    setIsPaused(false);
    setMediaReady(false);
    progress.value = 0;
  }, [progress]);

  return {
    creatorIndex, storyIndex, isPaused, mediaReady, progress,
    currentGroup, currentStory,
    goToNextStory, goToPrevStory, goToNextCreator, goToPrevCreator,
    pause, resume, onMediaReady, onVideoEnd, openAt,
  };
}
```

**B. Create `components/stories/StoryViewer.tsx`:**

Full-screen modal viewer that composes StoryMedia, StoryProgressBar, and gesture handling.

Props:
```typescript
type StoryViewerProps = {
  visible: boolean;
  groups: StoryGroup[];
  initialGroupIndex: number;
  onClose: () => void;
  onStoryViewed?: (storyId: string) => void;
};
```

Implementation details:

1. **Gesture handling** using react-native-gesture-handler:
   - `Gesture.Tap()` -- tap left third = prev story, right two-thirds = next story (use `event.x` vs `SCREEN_WIDTH / 3`)
   - `Gesture.LongPress()` with `minDuration(200)` -- onStart calls `pause()`, onEnd calls `resume()`
   - Compose: `Gesture.Exclusive(longPress, tap)` so long press takes priority
   - `Gesture.Fling()` horizontal for creator swiping: `.direction(Directions.LEFT)` calls `goToNextCreator()`, `.direction(Directions.RIGHT)` calls `goToPrevCreator()`
   - Compose all: `Gesture.Race(flingLeft, flingRight, Gesture.Exclusive(longPress, tap))`

2. **Layout** (full screen, black background):
   - `<Modal visible={visible} animationType="fade" statusBarTranslucent>`
   - `<GestureDetector gesture={composed}>` wrapping entire view
   - `<StoryMedia>` filling the screen
   - Top overlay with safe area insets: `<StoryProgressBar>` + author row (avatar, name, time ago, close button)
   - Close button (X icon) at top right

3. **Author header row**: Show current creator's avatar, display name, and relative time of current story. Use Feather "x" icon for close button.

4. **Mark as viewed**: When `currentStory` changes and `onStoryViewed` is provided, call it with the story ID. Track viewed IDs in a `useRef<Set<string>>` to avoid duplicate calls.

5. **Call `openAt(initialGroupIndex)` on mount** (when visible becomes true).

6. **Styling**: Match the aesthetic from tapzi-barber's StoryViewer (dark overlay at top, white text, semi-transparent close button). Use NativeWind/Tailwind classes where possible, fall back to StyleSheet for animated styles.

7. **timeAgo helper**: Copy the Romanian time-ago function from tapzi-barber's StoryViewer (acum, min, ore, zile).

Do NOT use setTimeout anywhere -- all timing is via Reanimated withTiming callbacks.
